#=> class

  • 正常的class实现:

  • 
class Bird
  attr_accessor :name, :sex
  def initialize name
    @name = name
  end
  
  def self.fly
    puts "bird can fly"
  end
  
  def say
    puts "i am #{@name}"
  end
end

bird = Bird.new("didi")
bird.sex = "male"
Bird.fly
